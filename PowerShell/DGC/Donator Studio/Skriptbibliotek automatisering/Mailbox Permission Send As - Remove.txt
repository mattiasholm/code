$EmailAddress = "gemensam@domain.tld"
$Deligate = "adam.boll@domain.tld"



if ((Get-PSSnapin | where {$_.Name -match "Exchange.Management"}) -eq $null) { Add-PSSnapin Microsoft.Exchange.Management.* }

if ((Get-Module | where {$_.Name -match "ActiveDirectory"}) -eq $null) { Import-Module ActiveDirectory }

$ADUser1 = Get-ADUser -Filter {UserPrincipalName -eq $EmailAddress}

Remove-ADPermission $ADUser1.DistinguishedName -User $Deligate -AccessRights ExtendedRight -ExtendedRights "Send As" -Confirm:$False

Remove-ADPermission $ADUser1.DistinguishedName -User $Deligate -AccessRights ExtendedRight -ExtendedRights "Receive As" -Confirm:$False
